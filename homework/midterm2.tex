\subsection*{3.2 - Binomial and Hypergeometric Probabilities}
\begin{enumerate}
	\item[33.] Suppose that r of N chips are red. Divide the chips into three groups of sizes \(n_1, n_2\), and \(n_3\), where \(n_1 + n_2 + n_3 = N\). Generalize the hypergeometric distribution to find the probability that the first group contains \(r_1\) red chips, the second group \(r_2\) red chips, and the third group \(r_3\) red chips, where \(r_1 + r_2 + r_3 = r\).
\end{enumerate}

\subsection*{3.4 - Continuous Random Variables}
\begin{enumerate}
	\item[4.] For persons infected with a certain form of malaria, the length of time spent in remission is described by the continuous pdf \(f_Y (y) = \frac{1}{9} y^2, 0 \leq y \leq 3\), where Y is measured in years. What is the probability that a malaria patient's remission lasts longer than one year?
\end{enumerate}

\subsection*{3.5 - Expected Values}
\begin{enumerate}
	\item[33.] Grades on the last Economics 301 exam were not very good. Graphed, their distribution had a shape similar to the pdf
		\[
			f_Y(y) = \frac{1}{5000} (100 - y), 0 \leq y \leq 100
		\].

	As a way of “curving” the results, the professor announces that he will replace each person’s grade, Y, with a new grade, \(g(Y)\), where \(g(Y) = 10 \sqrt{Y}\). Will the professor's strategy be successful in raising the class average above 60?
\end{enumerate}

\subsection*{3.6 - The Variance}
\begin{enumerate}
	\item[5.] Use theorem 3.6.1 to find the variance of the random variable Y, where
		\[
			f_Y(y) = 3(1 - y)^2, 0 < y < 1
		\]

	Theorem 3.6.1 states that let W be any random variable, discrete or continuous, having mean \(\mu\) and for which \(E(W^2)\) is finite. Then
		\[
			\textrm{Var}(W) = \sigma^2 = E(W^2) - \mu^2
		\] 
\end{enumerate}
